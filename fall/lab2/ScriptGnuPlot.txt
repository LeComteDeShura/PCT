#!/usr/bin/gnuplot
set terminal png size 1920,1080 enhanced font 'Arial, 16'
set output 'runge.png'

set style line 1 linecolor rgb 'red' linetype 1 linewidth 2

set border linewidth 2
set key top left
set grid
set mytics 1
set format y "%.4f"
set xlabel "Processes"
set format x "%3.0f"
set ylabel "Speedup"
set rmargin 4
set tmargin 2
plot "runge.dat" using 1:2 title "Runge" with linespoints ls 1
